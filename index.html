Arrays & Sliding Window
1. Maximum Subarray — find contiguous subarray with largest sum.
Easy▼

Pattern: Kadane's Algorithm — running max vs reset to current element.

function maxSubArray(nums: number[]): number { let maxSum = nums[0], curr = nums[0]; for (let i = 1; i < nums.length; i++) { curr = Math.max(nums[i], curr + nums[i]); maxSum = Math.max(maxSum, curr); } return maxSum; } // [-2,1,-3,4,-1,2,1,-5,4] → 6 ([4,-1,2,1]) // Time: O(n) Space: O(1)
2. Best Time to Buy and Sell Stock — one transaction, maximise profit.
Easy▼
function maxProfit(prices: number[]): number { let minPrice = Infinity, maxProfit = 0; for (const price of prices) { minPrice = Math.min(minPrice, price); maxProfit = Math.max(maxProfit, price - minPrice); } return maxProfit; } // [7,1,5,3,6,4] → 5 Time: O(n) Space: O(1)
3. Longest Substring Without Repeating Characters.
Medium▼
function lengthOfLongestSubstring(s: string): number { const set = new Set<string>(); let left = 0, max = 0; for (let right = 0; right < s.length; right++) { while (set.has(s[right])) set.delete(s[left++]); set.add(s[right]); max = Math.max(max, right - left + 1); } return max; } // "abcabcbb" → 3 Time: O(n) Space: O(min(n,m))
4. Product of Array Except Self — without division, O(n).
Medium▼

Two-pass: prefix products left→right, then multiply suffix right→left.

function productExceptSelf(nums: number[]): number[] { const n = nums.length, res = new Array(n).fill(1); let prefix = 1; for (let i = 0; i < n; i++) { res[i] = prefix; prefix *= nums[i]; } let suffix = 1; for (let i = n-1; i >= 0; i--) { res[i] *= suffix; suffix *= nums[i]; } return res; } // [1,2,3,4] → [24,12,8,6] Time: O(n) Space: O(1)
Two Pointers
5. Container With Most Water.
Medium▼

Move the pointer with the smaller height — moving the taller one can only decrease area.

function maxArea(height: number[]): number { let l = 0, r = height.length-1, max = 0; while (l < r) { max = Math.max(max, Math.min(height[l],height[r]) * (r-l)); if (height[l] < height[r]) l++; else r--; } return max; } // Time: O(n) Space: O(1)
6. 3Sum — all unique triplets that sum to zero.
Medium▼
function threeSum(nums: number[]): number[][] { nums.sort((a,b) => a-b); const res: number[][] = []; for (let i = 0; i < nums.length-2; i++) { if (i > 0 && nums[i] === nums[i-1]) continue; let l = i+1, r = nums.length-1; while (l < r) { const sum = nums[i]+nums[l]+nums[r]; if (sum === 0) { res.push([nums[i],nums[l],nums[r]]); while (l<r && nums[l]===nums[l+1]) l++; while (l<r && nums[r]===nums[r-1]) r--; l++; r--; } else if (sum < 0) l++; else r--; } } return res; } // Time: O(n²) Space: O(1)
Linked List
7. Reverse Linked List — iteratively and recursively.
Easy▼
function reverseList(head: ListNode | null): ListNode | null { let prev: ListNode|null = null, curr = head; while (curr) { const next = curr.next; curr.next = prev; prev = curr; curr = next; } return prev; } // Time: O(n) Space: O(1)
8. Detect Cycle — Floyd's tortoise and hare.
Easy▼
function hasCycle(head: ListNode | null): boolean { let slow = head, fast = head; while (fast && fast.next) { slow = slow!.next; fast = fast.next.next; if (slow === fast) return true; } return false; } // Time: O(n) Space: O(1)
9. Merge K Sorted Lists — divide and conquer.
Hard▼
function mergeKLists(lists: Array<ListNode|null>): ListNode|null { while (lists.length > 1) { const merged: Array<ListNode|null> = []; for (let i = 0; i < lists.length; i += 2) merged.push(mergeTwoLists(lists[i], lists[i+1] ?? null)); lists = merged; } return lists[0] ?? null; } function mergeTwoLists(l1: ListNode|null, l2: ListNode|null) { const d = new ListNode(0); let c = d; while (l1 && l2) { if (l1.val <= l2.val) { c.next=l1; l1=l1.next; } else { c.next=l2; l2=l2.next; } c = c.next!; } c.next = l1 ?? l2; return d.next; } // Time: O(N log k)
Trees
10. Binary Tree Level Order Traversal (BFS).
Medium▼
function levelOrder(root: TreeNode|null): number[][] { if (!root) return []; const res: number[][] = [], q = [root]; while (q.length) { const sz = q.length, lvl: number[] = []; for (let i = 0; i < sz; i++) { const n = q.shift()!; lvl.push(n.val); if (n.left) q.push(n.left); if (n.right) q.push(n.right); } res.push(lvl); } return res; } // Time: O(n) Space: O(n)
11. Validate Binary Search Tree.
Medium▼

Pass min/max bounds recursively — don't just compare parent.

function isValidBST(n: TreeNode|null, min=-Infinity, max=Infinity): boolean { if (!n) return true; if (n.val <= min || n.val >= max) return false; return isValidBST(n.left,min,n.val) && isValidBST(n.right,n.val,max); } // Time: O(n) Space: O(h)
12. Lowest Common Ancestor of Binary Tree.
Medium▼
function lowestCommonAncestor(root: TreeNode|null, p: TreeNode, q: TreeNode): TreeNode|null { if (!root || root===p || root===q) return root; const l = lowestCommonAncestor(root.left,p,q); const r = lowestCommonAncestor(root.right,p,q); if (l && r) return root; return l ?? r; } // Time: O(n) Space: O(h)
Graphs
13. Number of Islands — flood fill DFS.
Medium▼
function numIslands(grid: string[][]): number { let count = 0; const dfs = (r:number, c:number) => { if (r<0||r>=grid.length||c<0||c>=grid[0].length||grid[r][c]!=='1') return; grid[r][c]='0'; dfs(r+1,c);dfs(r-1,c);dfs(r,c+1);dfs(r,c-1); }; for (let r=0;r<grid.length;r++) for (let c=0;c<grid[0].length;c++) if (grid[r][c]==='1') { dfs(r,c); count++; } return count; } // Time: O(m·n) Space: O(m·n)
14. Course Schedule — detect cycle with DFS topological sort.
Medium▼
function canFinish(n: number, prereqs: number[][]): boolean { const adj = Array.from({length:n},()=>[] as number[]); for (const [a,b] of prereqs) adj[b].push(a); const state = new Array(n).fill(0); // 0=unvisited 1=visiting 2=done const dfs = (node:number): boolean => { if (state[node]===1) return false; // cycle if (state[node]===2) return true; state[node]=1; for (const next of adj[node]) if (!dfs(next)) return false; state[node]=2; return true; }; for (let i=0;i<n;i++) if (!dfs(i)) return false; return true; } // Time: O(V+E)
Dynamic Programming
15. Climbing Stairs — Fibonacci in disguise.
Easy▼
function climbStairs(n: number): number { if (n <= 2) return n; let p2=1, p1=2; for (let i=3;i<=n;i++) { const c=p1+p2; p2=p1; p1=c; } return p1; } // Time: O(n) Space: O(1)
16. Coin Change — minimum coins (unbounded knapsack).
Medium▼
function coinChange(coins: number[], amount: number): number { const dp = new Array(amount+1).fill(Infinity); dp[0]=0; for (let a=1;a<=amount;a++) for (const c of coins) if (c<=a) dp[a]=Math.min(dp[a], dp[a-c]+1); return dp[amount]===Infinity ? -1 : dp[amount]; } // Time: O(amount·coins.length) Space: O(amount)
17. 0/1 Knapsack — maximum value with weight limit.
Medium▼
function knapsack(w: number[], v: number[], cap: number): number { const n=w.length; const dp=Array.from({length:n+1},()=>new Array(cap+1).fill(0)); for (let i=1;i<=n;i++) for (let c=0;c<=cap;c++) { dp[i][c]=dp[i-1][c]; if (w[i-1]<=c) dp[i][c]=Math.max(dp[i][c],dp[i-1][c-w[i-1]]+v[i-1]); } return dp[n][cap]; } // Time: O(n·W) Space: O(n·W)
18. Longest Increasing Subsequence (LIS).
Medium▼
function lengthOfLIS(nums: number[]): number { const dp = new Array(nums.length).fill(1); let max = 1; for (let i=1;i<nums.length;i++) { for (let j=0;j<i;j++) if (nums[j]<nums[i]) dp[i]=Math.max(dp[i],dp[j]+1); max=Math.max(max,dp[i]); } return max; } // [10,9,2,5,3,7,101,18] → 4 Time: O(n²)
Stack & Heap
19. Daily Temperatures — monotonic stack pattern.
Medium▼
function dailyTemperatures(temps: number[]): number[] { const res = new Array(temps.length).fill(0), stack: number[] = []; for (let i=0;i<temps.length;i++) { while (stack.length && temps[i]>temps[stack[stack.length-1]]) { const idx=stack.pop()!; res[idx]=i-idx; } stack.push(i); } return res; } // Time: O(n) Space: O(n)
20. Top K Frequent Elements — bucket sort O(n) trick.
Medium▼

Since max frequency ≤ n, use frequency-indexed buckets for O(n) solution.

Node.js & JavaScript
Walk me through the Node.js event loop and its phases.
Frequent▼

Node.js handles concurrency via the event loop which cycles through: Timers (setTimeout/setInterval) → Pending callbacks (deferred I/O) → Poll (new I/O events, blocks here if nothing scheduled) → Check (setImmediate) → Close (socket close callbacks). Between phases: process.nextTick drains first, then Promise microtasks.

What is a MongoDB index and when does it hurt performance?
Common▼

Indexes speed up reads but hurt write-heavy collections (every insert/update must update all indexes). Hurt also: too many indexes consume RAM, low-cardinality fields (booleans), multikey indexes on arrays with many elements. Rule: index what you query and sort on.
